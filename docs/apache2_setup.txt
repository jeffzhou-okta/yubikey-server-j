api.yubico.com:

<VirtualHost *>
        ServerName api.yubico.com
        ServerAdmin tech@yubico.com

        RewriteEngine On
        RewriteLog "/var/log/rewrite_log"
        RewriteLogLevel 3

        Options +FollowSymLinks
    RewriteRule ^/verify(.*)$  http://localhost:8180/wsapi/verify$1%1 [P,L,QSA]

        ErrorLog /var/log/apache2/api.yubico.com-error.log
        LogLevel warn

        CustomLog /var/log/apache2/api.yubico.com-access.log combined
        ServerSignature On

</VirtualHost>






make sure to enable mod_rewrite, mod_proxy, and mod_proxy_http
